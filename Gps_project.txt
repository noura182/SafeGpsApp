## Implementasi Detail

### 1. LocationForegroundService.kt
Inti dari aplikasi — service yang **tidak bisa di-kill** walau app di-swipe.

```kotlin
@AndroidEntryPoint
class LocationForegroundService : LifecycleService() {

    private lateinit var fusedLocationClient: FusedLocationProviderClient
    private lateinit var locationCallback: LocationCallback
    private val repository: LocationRepository by inject()
    private val scope = CoroutineScope(Dispatchers.IO + SupervisorJob())

    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        super.onStartCommand(intent, flags, startId)

        when (intent?.action) {
            ACTION_START -> startTracking()
            ACTION_STOP  -> stopTracking()
        }

        // START_STICKY: service restart otomatis jika di-kill sistem
        return START_STICKY
    }

    private fun startTracking() {
        startForeground(NOTIF_ID, buildNotification("GPS aktif — mengirim lokasi..."))
        setupLocationCallback()
        requestLocationUpdates()
    }

    private fun setupLocationCallback() {
        locationCallback = object : LocationCallback() {
            override fun onLocationResult(result: LocationResult) {
                val loc = result.lastLocation ?: return
                scope.launch {
                    repository.sendLocation(loc)
                    updateNotification("Terkirim: ${formatTime()} · ${loc.latitude}, ${loc.longitude}")
                }
            }
        }
    }

    private fun requestLocationUpdates() {
        val interval = PreferencesManager.getInterval() // dalam ms

        // ✅ Kunci hemat baterai: PRIORITY_BALANCED_POWER_ACCURACY
        val request = LocationRequest.Builder(
            Priority.PRIORITY_BALANCED_POWER_ACCURACY,
            interval
        )
        .setMinUpdateIntervalMillis(interval / 2)
        .setMaxUpdateDelayMillis(interval + 60_000) // toleransi 1 menit
        .setWaitForAccurateLocation(false)          // jangan tunggu GPS fix sempurna
        .build()

        fusedLocationClient.requestLocationUpdates(
            request,
            locationCallback,
            Looper.getMainLooper()
        )
    }

    private fun buildNotification(text: String): Notification {
        val stopIntent = PendingIntent.getService(
            this, 0,
            Intent(this, LocationForegroundService::class.java).apply {
                action = ACTION_STOP
            },
            PendingIntent.FLAG_IMMUTABLE
        )

        return NotificationCompat.Builder(this, CHANNEL_ID)
            .setContentTitle("GPS Tracker")
            .setContentText(text)
            .setSmallIcon(R.drawable.ic_location)
            .setOngoing(true)             // tidak bisa di-dismiss user
            .setSilent(true)              // tidak bunyi/getar
            .setPriority(NotificationCompat.PRIORITY_LOW)
            .addAction(R.drawable.ic_stop, "Stop", stopIntent)
            .build()
    }

    override fun onDestroy() {
        super.onDestroy()
        fusedLocationClient.removeLocationUpdates(locationCallback)
        scope.cancel()
    }

    companion object {
        const val ACTION_START = "ACTION_START"
        const val ACTION_STOP  = "ACTION_STOP"
        const val NOTIF_ID     = 1001
        const val CHANNEL_ID   = "gps_tracker_channel"
    }
}
```

### 2. LocationRepository.kt
Kirim data ke Firestore dengan **batching & retry** hemat baterai.

```kotlin
class LocationRepository @Inject constructor(
    private val firestore: FirebaseFirestore,
    private val auth: FirebaseAuth
) {
    // Buffer lokal — kirim sekaligus jika offline
    private val pendingQueue = mutableListOf<LocationData>()

    suspend fun sendLocation(location: Location) {
        val userId = auth.currentUser?.uid ?: return

        val data = LocationData(
            lat       = location.latitude,
            lng       = location.longitude,
            accuracy  = location.accuracy,
            speed     = location.speed,
            timestamp = FieldValue.serverTimestamp(),
            deviceId  = Build.ID
        )

        try {
            firestore
                .collection("users")
                .document(userId)
                .collection("locations")
                .add(data)
                .await()
        } catch (e: Exception) {
            // Simpan lokal jika offline, kirim nanti
            pendingQueue.add(data)
            flushQueueWhenOnline()
        }
    }

    // Flush otomatis saat koneksi pulih (Firestore offline persistence)
    private fun flushQueueWhenOnline() {
        // Firestore sudah handle ini otomatis via offline cache
        // enableNetwork() akan trigger re-send pending writes
        firestore.enableNetwork()
    }
}

data class LocationData(
    val lat: Double = 0.0,
    val lng: Double = 0.0,
    val accuracy: Float = 0f,
    val speed: Float = 0f,
    val timestamp: Any? = null,
    val deviceId: String = ""
)
```

### 3. MainActivity.kt
UI minimal — satu toggle + status.

```kotlin
class MainActivity : AppCompatActivity() {

    private lateinit var binding: ActivityMainBinding

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)

        setupIntervalPicker()
        setupToggle()
        observeServiceState()
    }

    private fun setupToggle() {
        binding.switchGps.setOnCheckedChangeListener { _, isChecked ->
            val action = if (isChecked)
                LocationForegroundService.ACTION_START
            else
                LocationForegroundService.ACTION_STOP

            Intent(this, LocationForegroundService::class.java).also {
                it.action = action
                startForegroundService(it)  // Android 8+ wajib startForegroundService
            }
        }
    }

    private fun setupIntervalPicker() {
        val intervals = listOf(
            "5 menit"  to 5 * 60 * 1000L,
            "10 menit" to 10 * 60 * 1000L,
            "30 menit" to 30 * 60 * 1000L,
            "1 jam"    to 60 * 60 * 1000L
        )

        val adapter = ArrayAdapter(this, android.R.layout.simple_spinner_item,
            intervals.map { it.first })
        binding.spinnerInterval.adapter = adapter

        binding.spinnerInterval.onItemSelectedListener = object : AdapterView.OnItemSelectedListener {
            override fun onItemSelected(parent: AdapterView<*>, view: View?, pos: Int, id: Long) {
                PreferencesManager.saveInterval(intervals[pos].second)
                // Restart service dengan interval baru jika sedang aktif
                if (isServiceRunning()) restartService()
            }
            override fun onNothingSelected(parent: AdapterView<*>) {}
        }
    }
}
```

```xml
    android:name=".service.LocationForegroundService"
    android:foregroundServiceType="location"
    android:exported="false" />

    android:name=".receiver.BootReceiver"
```

## Strategi Hemat Baterai

| Teknik | Implementasi | Penghematan |
|--------|-------------|------------|
| `PRIORITY_BALANCED_POWER_ACCURACY` | Gunakan cell tower + WiFi, bukan GPS murni | ~60% vs GPS |
| `setWaitForAccurateLocation(false)` | Tidak menunggu fix GPS sempurna | Kurangi wake lock |
| `setMaxUpdateDelayMillis()` | Batching update — OS kirim sekaligus | Kurangi wake-up CPU |
| Interval minimal 5 menit | Tidak ada yang lebih pendek | GPS radio hemat |
| `PRIORITY_LOW` notification | Tidak ada suara/getar | Minor |
| Firestore offline persistence | Tidak retry saat offline | Hemat network |
| Single FusedLocation request | Satu callback, tidak polling | Tidak ada busy loop |
| Doze Mode compatible | FusedLocationProvider sudah handle | Otomatis |

## Build & Release via GitHub Actions

Karena EAS adalah untuk React Native, gunakan GitHub Actions untuk CI/CD yang setara:

### `.github/workflows/build-apk.yml`
```yaml
name: Build APK

on:
  push:
    branches: [main]
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup JDK 17
        uses: actions/setup-java@v4
        with:
          java-version: '17'
          distribution: 'temurin'

      - name: Setup Android SDK
        uses: android-actions/setup-android@v3

      - name: Add google-services.json
        run: echo '${{ secrets.GOOGLE_SERVICES_JSON }}' > app/google-services.json

      - name: Build Release APK
        run: ./gradlew assembleRelease

      - name: Sign APK
        uses: r0adkll/sign-android-release@v1
        with:
          releaseDirectory: app/build/outputs/apk/release
          signingKeyBase64: ${{ secrets.SIGNING_KEY }}
          alias: ${{ secrets.KEY_ALIAS }}
          keyStorePassword: ${{ secrets.KEY_STORE_PASSWORD }}
          keyPassword: ${{ secrets.KEY_PASSWORD }}

      - name: Upload APK Artifact
        uses: actions/upload-artifact@v4
        with:
          name: gps-tracker-release.apk
          path: app/build/outputs/apk/release/*.apk
```

```
App Launch
    │
    ├─► Firebase Auth (Login/Register)
    │
    ├─► Request ACCESS_FINE_LOCATION
    │       └─► Jika ditolak → tampilkan dialog edukasi
    │
    ├─► Request ACCESS_BACKGROUND_LOCATION  ← Android 11+ wajib step terpisah
    │       └─► Arahkan ke Settings > App > Permissions

    └─► User toggle GPS ON
```
